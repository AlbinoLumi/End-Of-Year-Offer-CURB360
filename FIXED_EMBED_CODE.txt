<div style="width: 100%; height: 100vh; min-height: 800px; overflow: hidden;">
  <iframe 
    src="https://end-of-year-offer-curb-360.vercel.app/" 
    width="100%" 
    height="100%" 
    frameborder="0" 
    scrolling="no"
    style="border: none; min-height: 800px; display: block;"
    title="CURB360 End of Year Offer"
    id="curb360-embed-fixed">
  </iframe>
</div>

<script>
(function() {
  var iframe = document.getElementById('curb360-embed-fixed');
  if (!iframe) return;
  
  function adjustHeight() {
    try {
      var iframeDoc = iframe.contentDocument || iframe.contentWindow.document;
      var body = iframeDoc.body;
      var html = iframeDoc.documentElement;
      
      var height = Math.max(
        body.scrollHeight,
        body.offsetHeight,
        html.clientHeight,
        html.scrollHeight,
        html.offsetHeight
      );
      
      iframe.style.height = height + 'px';
      iframe.parentElement.style.height = height + 'px';
    } catch (e) {
      // Cross-origin restriction - use viewport height
      iframe.style.height = '100vh';
      iframe.parentElement.style.height = '100vh';
    }
  }
  
  iframe.onload = function() {
    setTimeout(adjustHeight, 500);
    setTimeout(adjustHeight, 1500);
    setTimeout(adjustHeight, 3000);
  };
  
  window.addEventListener('resize', function() {
    setTimeout(adjustHeight, 100);
  });
})();
</script>
